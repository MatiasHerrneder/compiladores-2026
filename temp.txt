PROGRAM.SECTION

/* Declaraciones de variables */
DECVAR
  INT a, b, c
  FLOAT x
  HEX z
  STRING msg
ENDDECVAR

// Inicializaciones / asignaciones
a = 10
b = 20
c = 123
x = 3.14
z = 0hAE2
msg = "Hola mundo"

// Caso: comparación simple y llamada a la función especial
IF (a == b) THEN
  show "a y b son iguales"
ELSE
  // #Iguales toma una expresión y una lista de ids entre corchetes
  show #Iguales(a, [a, a, 10, 5+5])
ENDIF

// Uso de negación lógica y operadores combinados
IF  (!(a == b) && (x > 2.0)) THEN
  show "Condicion compuesta verdadera"
ENDIF

show z

// Bucle REPEAT ... UNTIL
REPEAT
  a = a + 1
  show a
UNTIL a > 12

show a

// Mostrar mensaje final
show msg

ENDPROGRAM.SECTION
